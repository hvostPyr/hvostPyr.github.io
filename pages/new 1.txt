class Pair<T1, T2> {
    private final T1 i value;
    private final T2 s value;
    
    public static T1 getFirst(){
        return this.i;
    }
    public static T2 getSecond(){
        return this.s;
    }
    
    public static <T1, T2> Pair<T1, T2> of(T1 first, T2 second) {
        return new Pair<>(first, second);
    }
    
    @Override
    public boolean equals(Object obj){
        try{
            Pair<> p = (Pair<>) obj;
            return (this.getFirst()==p.getFirst() && this.getSecond()==p.getSecond());
        }
        catch(Exception e){
            return false;
        }
    }
    
    @Override
    public int hashCode(){
        T1 t1 = this.getFirst();
        T2 t2 = this.getSecond();
        
        return t1.hashCode() + t2.hashCode();
    }
    
    

}
